If `_data/galleries.json` exists (generated by `scripts/generate_galleries.rb`), we can render from the precomputed index instead. {%- endcomment -%} {%- assign gallery_index = site.data.galleries.galleries | default: nil -%} {%- assign gallery_files = nil -%} {%- if gallery_index and include.folder -%} {%- assign gallery_files = gallery_index[include.folder] -%} {%- endif -%} {%- if gallery_files -%}

{%- for file in gallery_files -%} {{ file.filename }} {%- endfor -%}

{%- else -%}

{%- for file in site.static_files -%} {%- if file.path contains include.folder -%} {%- assign ext = file.extname | downcase -%} {%- if ext == '.jpg' or ext == '.jpeg' or ext == '.png' or ext == '.webp' or ext == '.gif' -%} {%- assign filenameparts = file.path | split: "/" -%} {%- assign filename = filenameparts | last | replace: file.extname,"" -%} {{ filename }} {%- endif -%} {%- endif -%} {%- endfor -%}

{%- endif -%}
